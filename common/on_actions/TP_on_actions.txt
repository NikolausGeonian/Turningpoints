on_actions = {

	on_startup = {
		effect = {
			every_country = {
				add_to_array = { #Burgundian Inheritance
					TP_achievement_array = 1
				}
				add_to_array = {
					TP_achievement_array = 2
				}
				add_to_array = {
					TP_achievement_array = 3
				}
				add_to_array = {
					TP_achievement_array = 4
				}
				add_to_array = {
					TP_achievement_array = 5
				}
				add_to_array = {
					TP_achievement_array = 6
				}
				add_to_array = {
					TP_achievement_array = 7
				}
				add_to_array = {
					TP_achievement_array = 8
				}
				add_to_array = {
					TP_achievement_array = 9
				}
				add_to_array = {
					TP_achievement_array = 10
				}
				add_to_array = {
					TP_achievement_array = 11
				}
			}
		}

		effect = {
			every_country = {
				limit = { 
                    national_dissent = yes
                }
                set_variable = { national_dissent = 0 }
                set_variable = { national_dissent_growth = 0.005 }
                set_variable = { national_dissent_reduction = -0.005 }
            }
		}

		effect = {
			every_country = {
				limit = { 
                    tag = GBR
                }
				add_to_variable = {
				    var = national_dissent
				    value = 0.1
				}
            }
		}

		effect = {
			every_country = {
				limit = { 
                    tag = FRA
                }
				add_to_variable = {
				    var = national_dissent
				    value = 0.1
				}
            }
		}
    on_daily = {
		effect = {
            if = {
				limit = { 
					check_variable = { national_dissent < 1 }
					OR = {
                    	has_stability < 0.5
						AND = {
							has_war = yes
							has_war_support < 0.5
						}
					}
                }
 
                add_to_variable = { national_dissent = national_dissent_growth }
            }
        }

		effect = {
            if = {
				limit = { 
					check_variable = { national_dissent < 1 }
					check_variable = { national_dissent > 0 }
                    has_stability > 0.5
                }
 
                add_to_variable = { national_dissent = national_dissent_reduction }
            }
        }
	}
}
