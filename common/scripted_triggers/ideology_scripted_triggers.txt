has_socialist_government = {
	custom_trigger_tooltip = {
		tooltip = TP_is_socialist
		OR = {
			has_government = democratic_socialism
			has_government = patriotic_socialism
			AND = {
				has_government = social_democratism
				has_country_flag = soc_dems_are_socialists
			}
		}
	}
}

not_has_socialist_government = {
	custom_trigger_tooltip = {
		tooltip = TP_is_not_socialist

			has_government = democratic_socialism
			has_government = patriotic_socialism
			AND = {
				has_government = social_democratism
				has_country_flag = soc_dems_are_socialists

		}
	}
